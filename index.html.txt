    <script>
        function startRace() {
            let runners = [
                document.getElementById("runner1"),
                document.getElementById("runner2"),
                document.getElementById("runner3")
            ];

            let winnerDeclared = false;
            document.getElementById("winner").innerText = "";

            runners.forEach(runner => {
                runner.style.left = "0px";

                let interval = setInterval(() => {
                    let current = parseInt(runner.style.left);
                    let move = Math.floor(Math.random() * 20);
                    runner.style.left = current + move + "px";

                    if (current >= 800 && !winnerDeclared) {
                        winnerDeclared = true;
                        document.getElementById("winner").innerText = "🎉 Winner!";
                        clearInterval(interval);
                    }
                }, 200);
            });
        }
    </script>
